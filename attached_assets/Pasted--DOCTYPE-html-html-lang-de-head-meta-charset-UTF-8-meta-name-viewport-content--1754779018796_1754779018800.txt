    // Jahr
    document.getElementById('y').textContent = new Date().getFullYear();

    // Formular – öffnet vorgefüllte WhatsApp, fällt auf Mailto zurück
    const form = document.getElementById('contactForm');
    const help = document.getElementById('formHelp');

    form.addEventListener('submit', (e)=>{
      e.preventDefault();
      const data = new FormData(form);
      const email = (data.get('email') || '').trim();
      const ign = (data.get('ign') || '').trim();
      const friend = (data.get('friend') || '').trim();
      const details = (data.get('details') || '').trim();

      if(!email || !ign || !friend){
        help.textContent = 'Bitte fülle E-Mail, Ingame-Name und Freundschaftslink aus.';
        return;
      }

      const msg = `Neue Sticker-Anfrage:%0A` +
                  `E-Mail: ${encodeURIComponent(email)}%0A` +
                  `IGN: ${encodeURIComponent(ign)}%0A` +
                  `Link: ${encodeURIComponent(friend)}%0A` +
                  `Sticker: ${encodeURIComponent(details || '-')}`;

      // Versuch WhatsApp
      const wa = `https://wa.me/4915237250453?text=${msg}`;
      window.open(wa, '_blank', 'noopener');

      // Optional: Fallback Mailto (Passe Empfänger an)
      const mail = `mailto:service@babixgo.de?subject=Sticker%20Bestellung&body=${msg}`;
      setTimeout(()=>window.location.href = mail, 400);

      help.textContent = 'Danke! WhatsApp/Mail wird geöffnet…';
      form.reset();
    }, {passive:false});
  </script>
